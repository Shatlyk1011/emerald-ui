<Tab value="Code">
```tsx
'use client'

import { useEffect, useRef } from 'react'
import { gsap } from 'gsap'

export function GSAPAnimatedCard() {
  const cardRef = useRef<HTMLDivElement>(null)
  const overlayRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    if (!cardRef.current || !overlayRef.current) return

    const card = cardRef.current
    const overlay = overlayRef.current

    // GSAP animation timeline
    const tl = gsap.timeline({ paused: true })
    
    tl.to(card, {
      scale: 1.05,
      rotateY: 5,
      duration: 0.3,
      ease: 'power2.out'
    })
    .to(overlay, {
      opacity: 1,
      duration: 0.3
    }, '<')

    // Event listeners
    const handleMouseEnter = () => tl.play()
    const handleMouseLeave = () => tl.reverse()

    card.addEventListener('mouseenter', handleMouseEnter)
    card.addEventListener('mouseleave', handleMouseLeave)

    return () => {
      card.removeEventListener('mouseenter', handleMouseEnter)
      card.removeEventListener('mouseleave', handleMouseLeave)
    }
  }, [])

  return (
    <div 
      ref={cardRef}
      className="relative w-80 h-48 bg-gradient-to-br from-purple-500 to-pink-500 rounded-2xl shadow-xl p-6 cursor-pointer overflow-hidden"
    >
      <div className="relative z-10">
        <h3 className="text-2xl font-bold text-white mb-2">GSAP Card</h3>
        <p className="text-white/90">Hover me to see the magic!</p>
      </div>
      <div 
        ref={overlayRef}
        className="absolute inset-0 bg-white/20 backdrop-blur-sm opacity-0"
      />
    </div>
  )
}
```

## Props

This component doesn't accept props in the basic version, but you can easily extend it to customize:

- Colors and gradients
- Animation duration and easing
- Card dimensions
- Content

## Customization

You can customize the GSAP timeline to create different effects:

```tsx
// Example: Add rotation on hover
tl.to(card, {
  scale: 1.05,
  rotate: 3,
  duration: 0.4,
  ease: 'elastic.out(1, 0.5)'
})
```

## Notes

- GSAP provides better performance than CSS transitions for complex animations
- The timeline approach makes it easy to sequence multiple animations
- Use `paused: true` to control when animations play
